		Matrix t(rows, cols, 0);
		size_t i1 = 0;
		size_t j1 = 0;

		for (size_t i = 0; i < rows; i++){
			for (size_t j = 0; j < cols; j++){
				t[i][j] = matrix[i1][j1];

		}
